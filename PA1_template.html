dataframe$date<-as.Date(as.character(dataframe$date),'%Y-%m-%d')
dataframe$weekdays<-weekdays(dataframe$date)
dataframe$category<-rep("weekday",dim(dataframe)[1])

        dataframe$category[i]<-"weekend"
